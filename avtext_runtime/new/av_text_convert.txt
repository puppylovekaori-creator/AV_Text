"""
av_text_convert.py

サクラエディタの長いマクロを置き換える用スクリプト（DB版）。
共通部品は avtext_common.py を利用する。

機能（従来互換ベース）
- 記号／空白の正規化
- F**K -> FUCK（それ以外の半角 * は全角 ＊ に）
- Windows 禁止文字を全角記号へ変換
- 別名 -> canonical display へ統一（DB/CSV）
- DB/CSV に登録されている名前の区切り補助
- 同名トークン重複整理
- 未登録名の DB/CSV 追加（1行目トークン除外モードあり）
- A(B) / (A)B からエイリアス抽出して DB/CSV 更新
- 行全体の完全2回繰り返し圧縮（例: 初愛ねんね初愛ねんね）
- 最終1行 + .mp4 + 改行で出力（cp932）

追加要件
- actress.txt（= registration_text）内の「？−？」行は無視する
  例:
    Aさん
    ？−？
    Cさん
  -> Aさん Cさん.mp4
"""

# ===================== 共通設定 =====================

MAX_FILENAME_LEN = 240  # 警告の目安（切り詰めはしない）
SCRIPT_DIR = Path(__file__).resolve().parent

# actress.txt 側で混ざることがある「不明行」を無視する
# 例: ？−？ / ?-? / ？-？ / ?－? など
_UNKNOWN_LINE_RE = re.compile(
    r"^\s*[?？]\s*[-‐-‒–—―ー－−]\s*[?？]\s*$"
)


# ===================== ユーティリティ =====================

def _normalize_for_output_safe(char_normalizer: CharNormalizer, text: str) -> str:
    if not text:
        return text
    fn = getattr(char_normalizer, "normalize_for_output", None)
    if callable(fn):
        return fn(text)
    return char_normalizer.normalize_for_match(text)


def is_unknown_placeholder_line(line: str) -> bool:
    if line is None:
        return False
    s = line.replace("\u3000", " ").strip()
    return bool(_UNKNOWN_LINE_RE.match(s))


def filter_registration_lines(registration_text: str) -> tuple[str, int]:
    """
    registration_text（2行目以降）から「？−？」系の行を除外する。
    改行区切りで判定し、他の行は維持。
    """
    if not registration_text:
        return registration_text, 0

    kept: list[str] = []
    skipped = 0
    for line in registration_text.splitlines():
        if is_unknown_placeholder_line(line):
            skipped += 1
            continue
        kept.append(line)

    return "\n".join(kept), skipped


def get_text_excluding_first_line(text: str) -> str:
    lines = text.splitlines()
    if len(lines) <= 1:
        return ""
    return "\n".join(lines[1:])


def dedupe_line_full_repeat(raw_text: str) -> str:
    """
    行全体が「同じ文字列2回連結」になっている場合だけ半分に圧縮。
    """
    lines = raw_text.splitlines()
    out_lines: list[str] = []

    for line in lines:
        s = line.strip()
        if not s:
            out_lines.append(line)
            continue

        # 空白や括弧入りは保守的に対象外
        if re.search(r"\s", s):
            out_lines.append(line)
            continue
        if any(ch in s for ch in "()（）／【】"):
            out_lines.append(line)
            continue

        n = len(s)
        if n % 2 == 0:
            half = s[: n // 2]
            if half * 2 == s:
                out_lines.append(half)
                continue

        out_lines.append(line)

    return "\n".join(out_lines)


def has_people_separators(text: str) -> bool:
    """
    registration_text が既に区切られているかの判定。
    """
    s = (text or "").strip()
    if not s:
        return False

    if "】・【" in s:
        return True
    if "／" in s or "/" in s:
        return True
    if "・" in s or "、" in s or "," in s:
        return True
    if "\t" in s:
        return True
    if "\n" in s or "\r" in s:
        return True
    if " " in s:
        return True

    return False


# ===================== トークン分割（av_text 固有） =====================

def insert_spaces_for_aliases(text: str, alias_names_sorted: Iterable[str]) -> str:
    """
    既知名の直後にだけスペース（最長一致greedy）。
    """
    if not text:
        return text

    cand_sorted = [n for n in alias_names_sorted if n]

    by_first: dict[str, list[str]] = {}
    for name in cand_sorted:
        by_first.setdefault(name[0], []).append(name)

    out: list[str] = []
    i = 0
    n = len(text)

    while i < n:
        ch = text[i]
        matched = None

        cand_list = by_first.get(ch)
        if cand_list:
            for name in cand_list:
                if text.startswith(name, i):
                    prev = text[i - 1] if i > 0 else ""
                    nextc = text[i + len(name)] if i + len(name) < n else ""

                    # 括弧の内側を壊しにくくする
                    if prev in ("(", "（"):
                        continue
                    if nextc in (")", "）", "(", "（"):
                        continue

                    matched = name
                    break

        if matched:
            out.append(matched)
            j = i + len(matched)
            if j < n:
                nxt = text[j]
                if (not nxt.isspace()) and (nxt not in (")", "）", "(", "（")):
                    out.append(" ")
            i += len(matched)
        else:
            out.append(ch)
            i += 1

    return "".join(out)


def insert_spaces_around_aliases_for_registration(
    text: str,
    alias_names_sorted: Iterable[str],
    resolver: AliasResolver,
    char_normalizer: CharNormalizer,
) -> str:
    """
    registration_text 用の強め分割：
    既知名（old/new）＋括弧/空白なしの display を候補にして、前後にスペースを補助。
    """
    if not text:
        return text

    name_to_display = resolver.get_name_to_display_map()

    candidates: set[str] = set(n for n in alias_names_sorted if n)

    # display も保護候補に追加（括弧/空白なしのみ）
    for disp in set(name_to_display.values()):
        d = _normalize_for_output_safe(char_normalizer, (disp or "").strip())
        if not d:
            continue
        if re.search(r"[\s()（）]", d):
            continue
        candidates.add(d)

    cand_sorted = sorted(candidates, key=len, reverse=True)

    by_first: dict[str, list[str]] = {}
    for name in cand_sorted:
        by_first.setdefault(name[0], []).append(name)

    out: list[str] = []
    i = 0
    n = len(text)

    def is_ws(ch: str) -> bool:
        return bool(ch) and ch.isspace()

    while i < n:
        ch = text[i]
        matched = None

        cand_list = by_first.get(ch)
        if cand_list:
            for name in cand_list:
                if text.startswith(name, i):
                    prev = text[i - 1] if i > 0 else ""
                    nextc = text[i + len(name)] if i + len(name) < n else ""

                    if prev in ("(", "（"):
                        continue
                    if nextc in (")", "）", "(", "（"):
                        continue

                    matched = name
                    break

        if matched:
            if out:
                last = out[-1]
                if (not is_ws(last)) and (last not in ("(", "（")):
                    out.append(" ")

            out.append(matched)

            j = i + len(matched)
            if j < n:
                nxt = text[j]
                if (not is_ws(nxt)) and (nxt not in (")", "）", "(", "（")):
                    out.append(" ")

            i += len(matched)
        else:
            out.append(ch)
            i += 1

    return "".join(out)


def unify_aliases_tokens(tokens: list[str], resolver: AliasResolver) -> list[str]:
    unified: list[str] = []
    for tok in tokens:
        disp = resolver.resolve_display(tok)
        unified.append(disp if disp else tok)
    return unified


def dedupe_space_separated_tokens(tokens: list[str]) -> list[str]:
    seen: set[str] = set()
    out: list[str] = []
    for tok in tokens:
        if tok not in seen:
            seen.add(tok)
            out.append(tok)
    return out


def normalize_and_tokenize(
    raw: str,
    resolver: AliasResolver,
    text_normalizer: TextNormalizer,
    char_normalizer: CharNormalizer,
    *,
    for_registration: bool = False,
) -> tuple[str, list[str]]:
    """
    1ブロック分のテキストを整形してトークン化（av_text 固有ルール）。
    """
    alias_names_sorted = resolver.get_alias_names_sorted()

    text = _normalize_for_output_safe(char_normalizer, raw.strip())
    text = text.replace("（", "(").replace("）", ")")

    # A(B) / (A)B を先に display へ寄せる
    text = text_normalizer.canonicalize_parenthesized_alias_tokens(text, resolver)

    # 区切り・記号
    text = text.replace("／", " ")
    text = text.replace("】・【", " ")

    text = text.replace("\r\n", " ")
    text = text.replace("\r", " ")
    text = text.replace("\n", " ")

    text = text.replace("/", " ")
    text = text.replace(":", "：")
    text = text.replace(".", "")
    text = text.replace("　", " ")

    # 既知名の分割補助
    if for_registration:
        text = insert_spaces_around_aliases_for_registration(
            text,
            alias_names_sorted,
            resolver,
            char_normalizer,
        )
    else:
        text = insert_spaces_for_aliases(text, alias_names_sorted)

    # F**K 特殊扱い + 残り * 正規化
    text = text.replace("F**K", "FUCK")
    text = text.replace("*", "＊")

    # カンマは区切り
    text = text.replace(",", " ")

    text = re.sub(r"[ ]+", " ", text).strip()

    tokens = text.split()
    tokens = unify_aliases_tokens(tokens, resolver)
    tokens = [t for t in tokens if t != "#"]
    tokens = dedupe_space_separated_tokens(tokens)

    text = " ".join(tokens)
    return text, tokens


# ===================== alias pair 検出 / 新規登録判定 =====================

def detect_alias_pairs_in_raw(raw_text: str, char_normalizer: CharNormalizer) -> list[tuple[str, str]]:
    """
    registration_text の生テキストから A(B) / (A)B を抽出。
    """
    raw_text = _normalize_for_output_safe(char_normalizer, raw_text)

    pat1 = re.compile(r"([^\s()（）]+?)[ \t]*[\(（][ \t]*([^\(\)（）]+?)[ \t]*[\)）]")
    pat2 = re.compile(r"[\(（][ \t]*([^\(\)（）]+?)[ \t]*[\)）][ \t]*([^\s()（）]+)")

    pairs: list[tuple[str, str]] = []

    for m in pat1.finditer(raw_text):
        a = _normalize_for_output_safe(char_normalizer, m.group(1).strip())
        b = _normalize_for_output_safe(char_normalizer, m.group(2).strip())
        if a and b:
            pairs.append((a, b))

    for m in pat2.finditer(raw_text):
        a = _normalize_for_output_safe(char_normalizer, m.group(1).strip())
        b = _normalize_for_output_safe(char_normalizer, m.group(2).strip())
        if a and b:
            pairs.append((a, b))

    return pairs


def is_plausible_name_token(tok: str) -> bool:
    if not tok:
        return False
    if len(tok) >= 25:
        return False
    bad_chars = set('「」『』【】"“”…')
    if any(ch in tok for ch in bad_chars):
        return False
    if any(ch in tok for ch in "()（）"):
        return False
    return True


# ===================== 変換本体 =====================

def convert_text(
    raw_text: str,
    first_line_raw: str,
    registration_text: str,
    *,
    resolver: AliasResolver,
    text_normalizer: TextNormalizer,
    char_normalizer: CharNormalizer,
    logger: SimpleLogger,
    exclude_first_line_tokens_for_new_names: bool,
) -> tuple[str, list[str], list[tuple[str, str]]]:
    """
    戻り値:
      (最終出力テキスト, new_names, alias_pairs)
    """
    known_tokens = resolver.get_known_tokens()

    # 比較用に normalize_for_match の集合も用意
    known_token_keys = {
        char_normalizer.normalize_for_match(k)
        for k in known_tokens
        if k
    }

    people_is_already_separated = has_people_separators(registration_text)
    use_strong_split_for_people = not people_is_already_separated

    # ファイル名生成用：1行目と2行目以降を別々に正規化してから連結
    tokens_first: list[str] = []
    tokens_rest: list[str] = []

    if first_line_raw:
        _, tokens_first = normalize_and_tokenize(
            first_line_raw,
            resolver,
            text_normalizer,
            char_normalizer,
            for_registration=False,
        )

    if registration_text:
        _, tokens_rest = normalize_and_tokenize(
            registration_text,
            resolver,
            text_normalizer,
            char_normalizer,
            for_registration=use_strong_split_for_people,
        )

    tokens_all = tokens_first + tokens_rest

    if not tokens_all:
        _, tokens_all = normalize_and_tokenize(
            raw_text,
            resolver,
            text_normalizer,
            char_normalizer,
            for_registration=False,
        )

    # 登録用トークン（registration_text だけ）
    _, tokens_reg = normalize_and_tokenize(
        registration_text,
        resolver,
        text_normalizer,
        char_normalizer,
        for_registration=use_strong_split_for_people,
    )

    # 1行目トークン（新規名除外判定用）
    first_line_tokens: set[str] = set()
    first_line_token_keys: set[str] = set()
    if first_line_raw:
        _, first_tokens = normalize_and_tokenize(
            first_line_raw,
            resolver,
            text_normalizer,
            char_normalizer,
            for_registration=False,
        )
        first_line_tokens = set(first_tokens)
        first_line_token_keys = {
            char_normalizer.normalize_for_match(t) for t in first_tokens if t
        }

    # alias pairs（registration_text のみ）
    alias_pairs = detect_alias_pairs_in_raw(registration_text, char_normalizer)
    alias_pair_name_keys: set[str] = set()
    for a, b in alias_pairs:
        if a:
            alias_pair_name_keys.add(char_normalizer.normalize_for_match(a))
        if b:
            alias_pair_name_keys.add(char_normalizer.normalize_for_match(b))

    # 新規名抽出
    new_names: list[str] = []
    seen_new_keys: set[str] = set()

    for tok in tokens_reg:
        tok_out = _normalize_for_output_safe(char_normalizer, tok)
        tok_key = char_normalizer.normalize_for_match(tok_out)

        if not is_plausible_name_token(tok_out):
            continue
        if tok_key in known_token_keys:
            continue
        if exclude_first_line_tokens_for_new_names and tok_key in first_line_token_keys:
            continue
        if tok_key in alias_pair_name_keys:
            continue
        if tok_key in seen_new_keys:
            continue

        seen_new_keys.add(tok_key)
        new_names.append(tok_out)

    base_line = " ".join(tokens_all)
    base_line = text_normalizer.sanitize_for_windows_filename(
        base_line,
        max_len=MAX_FILENAME_LEN,
        truncate=False,
        warn_only=True,
        logger=logger,
    )

    if not base_line.lower().endswith(".mp4"):
        base_line += ".mp4"

    if len(base_line) > MAX_FILENAME_LEN:
        logger.log(
            f"[WARN] filename length over threshold after ext: "
            f"len={len(base_line)} > {MAX_FILENAME_LEN} (no truncation)"
        )

    final_text = base_line + "\n"
    logger.log(f"[INFO] people_is_already_separated={people_is_already_separated}")
    logger.log(f"[INFO] registration_text_len={len(registration_text)}")
    return final_text, new_names, alias_pairs


# ===================== DB / CSV 更新（av_text 固有） =====================

def append_new_names_to_csv(
    csv_path: Path,
    new_names: list[str],
    char_normalizer: CharNormalizer,
    logger: SimpleLogger,
) -> None:
    if not new_names:
        return

    unique: list[str] = []
    seen_keys: set[str] = set()

    for n in new_names:
        n_out = _normalize_for_output_safe(char_normalizer, (n or "").strip())
        if not n_out:
            continue
        key = char_normalizer.normalize_for_match(n_out)
        if key in seen_keys:
            continue
        seen_keys.add(key)
        unique.append(n_out)

    if not unique:
        return

    if csv_path.exists():
        mode = "a"
        write_header = False
    else:
        mode = "w"
        write_header = True

    with csv_path.open(mode, encoding="utf-8-sig", newline="") as f:
        fieldnames = ["old", "new", "display"]
        writer = csv.DictWriter(f, fieldnames=fieldnames)
        if write_header:
            writer.writeheader()
        for name in unique:
            writer.writerow({"old": name, "new": "", "display": ""})

    logger.log(f"[INFO] {len(unique)} new names appended to CSV.")


def append_new_names_to_db(
    conn,
    new_names: list[str],
    char_normalizer: CharNormalizer,
    logger: SimpleLogger,
) -> None:
    """
    ACTRESS_ID は明示しない（テーブル側の DEFAULT を使う前提）。
    """
    if not new_names or conn is None:
        return

    unique: list[str] = []
    seen_keys: set[str] = set()

    for n in new_names:
        n_out = _normalize_for_output_safe(char_normalizer, (n or "").strip())
        if not n_out:
            continue
        key = char_normalizer.normalize_for_match(n_out)
        if key in seen_keys:
            continue
        seen_keys.add(key)
        unique.append(n_out)

    if not unique:
        return

    try:
        cur = conn.cursor()
        for name in unique:
            # 厳密一致＋異体字吸収の双方を考慮して既存確認
            cur.execute("SELECT OLD_NAME FROM dbo.ACTRESS_DATA")
            exists = False
            for row in cur.fetchall():
                old_name = (row[0] or "").strip()
                if char_normalizer.normalize_for_match(old_name) == char_normalizer.normalize_for_match(name):
                    exists = True
                    break

            if not exists:
                cur.execute(
                    "INSERT INTO dbo.ACTRESS_DATA (OLD_NAME) VALUES (?)",
                    (name,),
                )

        conn.commit()
        logger.log(f"[INFO] {len(unique)} new names inserted into DB.")
    except Exception as e:
        logger.log(f"[ERROR] append_new_names_to_db failed: {e!r}")


def update_alias_pairs_in_csv(
    csv_path: Path,
    alias_pairs: list[tuple[str, str]],
    char_normalizer: CharNormalizer,
    logger: SimpleLogger,
) -> None:
    if not alias_pairs:
        return

    unique_pairs: list[tuple[str, str]] = []
    seen_pair_keys: set[tuple[str, str]] = set()

    for a, b in alias_pairs:
        a_out = _normalize_for_output_safe(char_normalizer, (a or "").strip())
        b_out = _normalize_for_output_safe(char_normalizer, (b or "").strip())
        if not a_out or not b_out:
            continue

        a_key = char_normalizer.normalize_for_match(a_out)
        b_key = char_normalizer.normalize_for_match(b_out)
        key = tuple(sorted((a_key, b_key)))

        if key in seen_pair_keys:
            continue
        seen_pair_keys.add(key)
        unique_pairs.append((a_out, b_out))

    if not unique_pairs:
        return

    rows: list[dict[str, str]] = []
    fieldnames = ["old", "new", "display"]

    if csv_path.exists():
        with csv_path.open("r", encoding="utf-8-sig", newline="") as f:
            reader = csv.DictReader(f)
            if reader.fieldnames:
                fieldnames = list(reader.fieldnames)
            for row in reader:
                rows.append(
                    {
                        "old": _normalize_for_output_safe(char_normalizer, (row.get("old") or "").strip()),
                        "new": _normalize_for_output_safe(char_normalizer, (row.get("new") or "").strip()),
                        "display": _normalize_for_output_safe(char_normalizer, (row.get("display") or "").strip()),
                    }
                )

    index_by_old_key: dict[str, int] = {}
    for i, row in enumerate(rows):
        old = row.get("old", "")
        if not old:
            continue
        old_key = char_normalizer.normalize_for_match(old)
        if old_key and old_key not in index_by_old_key:
            index_by_old_key[old_key] = i

    def decide_old_new(a: str, b: str) -> tuple[str, str]:
        a_key = char_normalizer.normalize_for_match(a)
        b_key = char_normalizer.normalize_for_match(b)
        in_a = a_key in index_by_old_key
        in_b = b_key in index_by_old_key

        if in_a and not in_b:
            return a, b
        if in_b and not in_a:
            return b, a

        if len(a) < len(b):
            return a, b
        if len(b) < len(a):
            return b, a

        return a, b

    for a, b in unique_pairs:
        old_name, new_name = decide_old_new(a, b)
        old_key = char_normalizer.normalize_for_match(old_name)

        if old_key in index_by_old_key:
            row = rows[index_by_old_key[old_key]]
            row["new"] = new_name
        else:
            rows.append({"old": old_name, "new": new_name, "display": ""})
            index_by_old_key[old_key] = len(rows) - 1

    with csv_path.open("w", encoding="utf-8-sig", newline="") as f:
        writer = csv.DictWriter(f, fieldnames=fieldnames)
        writer.writeheader()
        for row in rows:
            writer.writerow(
                {
                    "old": row.get("old", ""),
                    "new": row.get("new", ""),
                    "display": row.get("display", ""),
                }
            )

    logger.log(f"[INFO] {len(unique_pairs)} alias pairs merged into CSV.")

            continue

        seen_pair_keys.add(key)
        unique_pairs.append((a_out, b_out))

    if not unique_pairs:
        return

    try:
        cur = conn.cursor()

        def find_row_by_old_name_fuzzy(name: str):
            """
            OLD_NAME の異体字差も吸収して既存行を探す。
            戻り値: (ACTRESS_ID, OLD_NAME, NEW_NAME) or None
            """
            target_key = char_normalizer.normalize_for_match(name)
            cur.execute("SELECT ACTRESS_ID, OLD_NAME, NEW_NAME FROM dbo.ACTRESS_DATA")
            for actress_id, old_name, new_name in cur.fetchall():
                old_name = (old_name or "").strip()
                if char_normalizer.normalize_for_match(old_name) == target_key:
                    return actress_id, old_name, (new_name or "")
            return None

        def decide_old_new(a: str, b: str) -> tuple[str, str]:
            row_a = find_row_by_old_name_fuzzy(a)
            row_b = find_row_by_old_name_fuzzy(b)

            if row_a and not row_b:
                return a, b
            if row_b and not row_a:
                return b, a

            if not row_a and not row_b:
                if len(a) < len(b):
                    return a, b
                if len(b) < len(a):
                    return b, a
                return a, b

            return a, b

        for a, b in unique_pairs:
            old_name, new_name = decide_old_new(a, b)

            row = find_row_by_old_name_fuzzy(old_name)
            if row:
                actress_id, _old_existing, current_new = row
                current_new = _normalize_for_output_safe(char_normalizer, (current_new or "").strip())
                if (not current_new) or (
                    char_normalizer.normalize_for_match(current_new)
                    != char_normalizer.normalize_for_match(new_name)
                ):
                    cur.execute(
                        "UPDATE dbo.ACTRESS_DATA "
                        "SET NEW_NAME = ?, UPDATED_AT = SYSDATETIME() "
                        "WHERE ACTRESS_ID = ?",
                        (new_name, actress_id),
                    )
            else:
                cur.execute(
                    "INSERT INTO dbo.ACTRESS_DATA (OLD_NAME, NEW_NAME) VALUES (?, ?)",
                    (old_name, new_name),
                )

        conn.commit()
        logger.log(f"[INFO] {len(unique_pairs)} alias pairs merged into DB.")
    except Exception as e:
        logger.log(f"[ERROR] update_alias_pairs_in_db failed: {e!r}")


# ===================== main =====================

def main() -> None:
    log_path = SCRIPT_DIR / "av_text_convert.log"
    aliases_csv = SCRIPT_DIR / "aliases.csv"
    setting_ini = SCRIPT_DIR / "setting.ini"

    logger = SimpleLogger(log_path)
    logger.log("=== av_text_convert start ===")

    file_io = FileTextIO()
    char_normalizer = CharNormalizer()
    text_normalizer = TextNormalizer()

    # DB接続（失敗時CSVフォールバック）
    conn = None
    try:
        cfg = DbConfigLoader.load_from_ini(setting_ini)
        if cfg:
            conn = DbConnector(logger).connect(cfg)
        else:
            logger.log("[INFO] DB config unavailable. Use CSV aliases.")
    except Exception as e:
        logger.log(f"[WARN] DB setup failed, fallback to CSV: {e!r}")
        conn = None

    # --- 入力ソース判定 ---
    is_arg_mode = len(sys.argv) >= 2

    if is_arg_mode:
        input_path = Path(sys.argv[1])
        if not input_path.exists():
            msg = f"入力ファイルが見つかりません: {input_path}"
            print(msg)
            logger.log(f"[ERROR] {msg}")
            try:
                if conn is not None:
                    conn.close()
            except Exception:
                pass
            return

        try:
            text, encoding_used = file_io.load_text_with_fallback(input_path)
        except UnicodeError:
            msg = "input ファイルの文字コードを utf-8 / cp932 で読めませんでした。"
            print(msg)
            logger.log(f"[ERROR] {msg}")
            try:
                if conn is not None:
                    conn.close()
            except Exception:
                pass
            return
        except Exception as e:
            msg = f"input ファイル読み込み失敗: {e!r}"
            print(msg)
            logger.log(f"[ERROR] {msg}")
            try:
                if conn is not None:
                    conn.close()
            except Exception:
                pass
            return

        logger.log(f"[INFO] input loaded: {input_path} (encoding={encoding_used})")

        preprocessed = dedupe_line_full_repeat(text)

        lines = preprocessed.splitlines()
        first_line_raw = lines[0] if lines else ""

        registration_text = get_text_excluding_first_line(preprocessed)
        registration_text, skipped_unknown_lines = filter_registration_lines(registration_text)
        if skipped_unknown_lines:
            logger.log(f"[INFO] skipped unknown placeholder lines in registration_text: {skipped_unknown_lines}")

        source_desc = f"{input_path} (encoding={encoding_used})"
        output_path = input_path.with_name(input_path.stem + "_converted" + input_path.suffix)

        # 引数ありモードは1行目トークンを新規名抽出から除外
        exclude_first_line_tokens_for_new_names = True

    else:
        title_path = SCRIPT_DIR / "title.txt"
        actress_path = SCRIPT_DIR / "actress.txt"

        if not title_path.exists():
            msg = f"title.txt が見つかりません: {title_path}"
            print(msg)
            logger.log(f"[ERROR] {msg}")
            try:
                if conn is not None:
                    conn.close()
            except Exception:
                pass
            return

        if not actress_path.exists():
            msg = f"actress.txt が見つかりません: {actress_path}"
            print(msg)
            logger.log(f"[ERROR] {msg}")
            try:
                if conn is not None:
                    conn.close()
            except Exception:
                pass
            return

        try:
            title_text, enc_title = file_io.load_text_with_fallback(title_path)
        except UnicodeError:
            msg = "title.txt の文字コードを utf-8 / cp932 で読めませんでした。"
            print(msg)
            logger.log(f"[ERROR] {msg}")
            try:
                if conn is not None:
                    conn.close()
            except Exception:
                pass
            return
        except Exception as e:
            msg = f"title.txt 読み込み失敗: {e!r}"
            print(msg)
            logger.log(f"[ERROR] {msg}")
            try:
                if conn is not None:
                    conn.close()
            except Exception:
                pass
            return

        try:
            actress_text, enc_act = file_io.load_text_with_fallback(actress_path)
        except UnicodeError:
            msg = "actress.txt の文字コードを utf-8 / cp932 で読めませんでした。"
            print(msg)
            logger.log(f"[ERROR] {msg}")
            try:
                if conn is not None:
                    conn.close()
            except Exception:
                pass
            return
        except Exception as e:
            msg = f"actress.txt 読み込み失敗: {e!r}"
            print(msg)
            logger.log(f"[ERROR] {msg}")
            try:
                if conn is not None:
                    conn.close()
            except Exception:
                pass
            return

        logger.log(f"[INFO] title loaded:  {title_path} (encoding={enc_title})")
        logger.log(f"[INFO] actress loaded: {actress_path} (encoding={enc_act})")

        title_lines = title_text.splitlines()
        title_line = title_lines[0].rstrip("\r\n") if title_lines else ""

        combined = title_line + "\n" + actress_text
        preprocessed = dedupe_line_full_repeat(combined)

        lines = preprocessed.splitlines()
        first_line_raw = lines[0] if lines else ""

        registration_text = get_text_excluding_first_line(preprocessed)
        registration_text, skipped_unknown_lines = filter_registration_lines(registration_text)
        if skipped_unknown_lines:
            logger.log(f"[INFO] skipped unknown placeholder lines in actress/registration_text: {skipped_unknown_lines}")

        source_desc = f"title.txt({enc_title}) + actress.txt({enc_act})"
        output_path = SCRIPT_DIR / "conv_converted.txt"

        # actress.txt は明示リストなので1行目トークン除外なし
        exclude_first_line_tokens_for_new_names = False

    # --- エイリアス読み込み（DB優先） ---
    try:
        aliases = AliasRepository.load(conn, aliases_csv)
    except Exception as e:
        logger.log(f"[ERROR] alias load failed: {e!r}")
        aliases = []

    resolver = AliasResolver(aliases, logger, char_normalizer)
    resolver.build()

    # raw_text（fallback用）は「first_line + filtered registration」に寄せておく
    raw_text_for_convert = first_line_raw + ("\n" + registration_text if registration_text else "")

    # --- 変換本体 ---
    converted, new_names, alias_pairs = convert_text(
        raw_text=raw_text_for_convert,
        first_line_raw=first_line_raw,
        registration_text=registration_text,
        resolver=resolver,
        text_normalizer=text_normalizer,
        char_normalizer=char_normalizer,
        logger=logger,
        exclude_first_line_tokens_for_new_names=exclude_first_line_tokens_for_new_names,
    )

    # --- 出力（cp932固定） ---
    try:
        file_io.write_cp932(output_path, converted)
        logger.log(f"[INFO] output written: {output_path}")
    except Exception as e:
        logger.log(f"[ERROR] failed to write output: {e!r}")
        raise

    # --- DB/CSV 更新（av_text_convert のみ） ---
    if conn is not None:
        update_alias_pairs_in_db(conn, alias_pairs, char_normalizer, logger)
        append_new_names_to_db(conn, new_names, char_normalizer, logger)
        try:
            conn.close()
        except Exception:
            pass
    else:
        update_alias_pairs_in_csv(aliases_csv, alias_pairs, char_normalizer, logger)
        append_new_names_to_csv(aliases_csv, new_names, char_normalizer, logger)

    logger.log(f"[INFO] new_names={new_names}")
    logger.log("=== av_text_convert end ===")

    print(f"変換完了: {output_path} （input: {source_desc}, output: cp932, CRLF, .mp4付き）")


if __name__ == "__main__":
    main()
